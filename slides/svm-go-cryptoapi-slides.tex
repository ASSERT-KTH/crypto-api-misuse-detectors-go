\begin{frame}
    \frametitle{Problem Context}
    \begin{itemize}
        \setlength{\itemsep}{0.7em}
        \item Go is widely used in security-critical infrastructure, including certificate services, container platforms, and cloud tooling.
        \item Unlike Java, the current landscape of cryptographic misuse detection for Go has not been systematically characterized.
    \end{itemize}
\end{frame}

\begin{frame}[fragile]
    \frametitle{What Does a Go Crypto Misuse Look Like?}
    \footnotesize
    \textbf{Traefik, CVE-2025-66491.} Kubernetes ingress provider:
    \vspace{0.5em}
        \begin{minipage}{0.78\textwidth}
        \begin{lstlisting}
InsecureSkipVerify: strings.ToLower(
    ptr.Deref(cfg.ProxySSLVerify, "off")) == "on",
        \end{lstlisting}
        \end{minipage}
    \vspace{0.2em}
    \normalsize
    \begin{itemize}
        \setlength{\itemsep}{0.5em}
        \item The user sets the annotation \texttt{proxy-ssl-verify: "on"} expecting TLS verification to be enabled.
        \item The expression evaluates to \texttt{true}, so \texttt{InsecureSkipVerify} becomes \texttt{true} and verification is silently disabled.
        \item Compiles, passes tests, ships to production with the security guarantee inverted.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Evaluation Challenge}
    \begin{itemize}
        \setlength{\itemsep}{0.7em}
        \item Existing Go detectors do not operate over a common target: they differ in supported rules, reporting granularity, and execution robustness.
        \item Raw alert counts are therefore hard to compare: more findings may reflect broader rule coverage, noisier heuristics, or simply a different reporting unit.
        \item Tools may differ in coverage, granularity, or robustness. We need to understand what each tool detects, not just how much.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Prior Work}
    \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item Java already has comparative studies of crypto misuse detectors.
        \item Go also has detector work: \textit{CryptoGo} introduced the first Go detector, and \Gopher extended that line of work.
        \item What is still missing is a systematic comparison of the Go detectors currently available in practice.
    \end{itemize}
    \vspace{0.5em}
    \begin{block}{This paper}
        We provide the first comparative study of currently available Go crypto misuse detectors.
    \end{block}
\vspace{0.35em}
{\tiny Full citations are in the paper.}
\end{frame}

\begin{frame}
    \frametitle{Research Questions}
    \begin{itemize}
        \setlength{\itemsep}{0.7em}
        \item \textbf{RQ1.} What is the design space of state-of-the-art tools for detecting cryptographic API misuses in Go?
        \item \textbf{RQ2.} To what extent are crypto misuse detection tools for Go consistent with each other?
    \end{itemize}
    \vspace{0.5em}
    \begin{block}{Study Design}
        RQ1 compares tool scope, supported rules, and detection approach.

        RQ2 runs all tools on 328 security-relevant Go projects and compares execution coverage, findings, and overlap.
    \end{block}
    \vspace{0.4em}
    \begin{itemize}
        \setlength{\itemsep}{0.35em}
        \item Goal: establish what these tools detect, how they differ, and what follows for practice and evaluation.
    \end{itemize}
\end{frame}

\begin{frame}
    \frametitle{Manual Analysis: When Does Agreement Mean Truth?}
    \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item \textbf{Rule \#05.} 4 of 5 detections shared by three tools (missed by \Gopher) are in mock or example code. Overlap is not always a correctness signal.
        \item \textbf{Rule \#11.} 3 of 5 sampled \Gosec-only detections are false positives. \SnykCode-only sample: no clear false positives.
        \item \textbf{Rule \#13.} All 5 detections shared across tools are true positives. \CodeQL uniquely catches custom callbacks beyond known patterns.
    \end{itemize}

\begin{frame}[fragile]
    \frametitle{What Goes Wrong: Three Examples}
    \footnotesize
    \vspace{-0.3em}

    \textbf{1. Same call, different meaning} \hfill \textit{Rule \#05, flagged by \Gopher}
    \begin{verbatim}
    h := blake2b.New512(nil)   // bug if used as a MAC
                               // fine for content hashing
    \end{verbatim}
    Static analysis cannot tell which one this is.
    \vspace{0.4em}

    \textbf{2. Pattern match misses the equivalent} \hfill \textit{Rule \#13, only \CodeQL}
    \begin{verbatim}
    cfg.HostKeyCallback = func(...) error { return nil }
    \end{verbatim}
    \Gosec and \Gopher look for \texttt{InsecureIgnoreHostKey} by name and miss this.
    \vspace{0.4em}

    \textbf{3. Field name flagged, value ignored} \hfill \textit{Rule \#11, \Gosec false positive}
    \begin{verbatim}
    tls.Config{MinVersion: tls.VersionTLS12}   // safe
    \end{verbatim}
    \Gosec sees \texttt{MinVersion} and fires regardless of the value.
\end{frame}

\begin{frame}
    \frametitle{Implications}
    \begin{itemize}
        \setlength{\itemsep}{0.75em}
        \item \textbf{For practitioners:} no single detector is enough. The tools miss different things, so relying on one leaves systematic blind spots.
        \item \textbf{For evaluators:} raw counts mislead. On insecure PRNG detection, \Gosec reports 2,517 findings where \Gopher reports 0, on the same code and for the same rule.
        \item \textbf{For tool builders:} the gap is qualitative, not only quantitative. Better detectors need shared misuse definitions and more program context.
    \end{itemize}
    \vspace{0.45em}
    \begin{block}{Closing claim}
        Detector choice changes both what practitioners see in code review and what researchers conclude in evaluation.
    \end{block}
\end{frame}

\begin{frame}
    \frametitle{Future Work}
    \begin{itemize}
        \setlength{\itemsep}{0.55em}
        \item \textbf{Shared definitions of misuse}. Without ground-truth benchmarks that tool authors agree on, the field cannot measure whether detectors are actually improving.
        \item \textbf{Context-aware detection}. Tools need to reason about how a value is used, not just which API produced it.
    \end{itemize}
\end{frame}
